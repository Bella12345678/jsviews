16 Codeless tags - so use custom helper tags for computed parameters

TitleLanguages

HTML:

<td>
	{{#each Languages}}
		{{=Name}}
		{{#nextToLast}}
			and
		{{else {{notLast}} }}
			, 
		{{/nextToLast}}
	{{/each}}
</td>

Script:

$.views.registerTags({

	notLast: function() {
        return this.itemNumber !== this.parent.data.length;
    },

	nextToLast: function( content ) {
		...
		$.render( content, this.data );
	}
});
