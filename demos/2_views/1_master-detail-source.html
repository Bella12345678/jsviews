Data Link: Linking the master-list selected data to the detail view

Script:

var app = { // Our app state - available for data linking.  
		selectedItem: null
	};

$( "#movieList" )
	.html(
		$( "#movieTemplate" ).render( movies )
	)
	.dataLink( movies ) // Data link movies to the movielist rendered HTML

	.delegate( "tr", "click", function () {
		select( $.tmplItem( this )); // Click finds the clicked template item, passes it to select()
	})

function select( tmplItem ) {
	if ( app.selectedItem ) {
		$( app.selectedItem.nodes ).removeClass( "selected");
	}
	if (tmplItem) {
		$( tmplItem.nodes ).addClass( "selected");
	}
	$.setField( app, "selectedItem", tmplItem ); // Set selectedItem as observable event
}

$.dataLink( app, function( event, args ) { // Data link to app and listen for selectedItem objectChange event
	if ( args.path === "selectedItem" ) {
		var selectedData = args.value.data;
		
		// Render the detail view for the selectedData
		$( "#movieDetail" )
			.html(
				$( "#detailViewTemplate" ).render( selectedData )
			);
	}
});
